#+BEGIN_SRC [c-mode]
main(){
    while(input != "quit"){
        parse(input);
        execute(input);
    }
}
execute(input){
    if(fork < 0)
        failed to fork;
    else if(fork > 0)
        wait();
    else
        if(input matches builtin function){
            call builtin function;
        }else if(execvp(input) >= 0){
            process executed;
        else
            process failed to execute;
    }
}
char[][]  parse(input){
    return 2d array of input, split at whitespace;
}
